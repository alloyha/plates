\documentclass[
	% -- opcos da classe memoir --
	12pt,				% tamanho da fonte
	openright,			% capitulos comecam em pag impar (insere pagina vazia caso preciso)
	twoside,			% para impressao em verso e verso. Oposto a oneside
	a4paper,			% tamanho do papel. 	
	% -- opcoes da classe abntex2 --
	%chapter=TITLE,		% titulos de capiiulos convertidos em letras maiusculas
	%section=TITLE,		% titulos de sessoes convertidos em letras maiusculas
	%subsection=TITLE,	% titulos de subsecoes convertidos em letras maiusculas
	%subsubsection=TITLE,% titulos de subsubsessoes convertidos em letras maiusculas
	% -- opcoes do pacote babel --
	english,			% idioma adicional para hifenizacao
	french,				% idioma adicional para hifenizacao
	spanish,			% idioma adicional para hifenizacao
	brazil				% o ultimo idioma e' o principal do documento
	]{abntex2}

% Inicializacao de Pacotes
\input{preambulo/Pacotes}

% Configuracao de pacotes 
\input{preambulo/PacotesConfig}

% Capa e Folha de Rosto
\input{preambulo/DadosDoTexto}

% Configuracos de aparencia do PDF final
\input{preambulo/PDFConfig}

% ---
% compila o indice
% ---
\makeindex
% ---

% ----
% Inicio do documento
% ----
\begin{document}

% ----------------------------------------------------------
% Elementos pre-textuais
% ----------------------------------------------------------

\input{texto/pretextual/Capa}

% Confirmaca do Trabalho de Conclusao de Curso
\input{texto/pretextual/Confirmacao}

% Resumo
\input{texto/pretextual/Resumo}

% Indice e Tabelas
\input{texto/pretextual/TabelaDeIndices}

% ----------------------------------------------------------
% Elementos pre-textuais
% ----------------------------------------------------------
\textual

% Introducao
\input{texto/textual/Introducao}

% Revisao Bibliografica
\input{texto/textual/RevisaoBibliografica} 

% Desenvolvimento
\input{texto/textual/Desenvolvimento}

% Conclusoes
\input{texto/textual/Agenda}


% Anexos
\input{texto/bibliografia/Bibliografia}


% ----------------------------------------------------------
% Elementos pos-textuais
% ----------------------------------------------------------
\postextual

% Apendices
\input{texto/postextual/Apendices}

% Anexos
\input{texto/postextual/Anexos}

\end{document}
